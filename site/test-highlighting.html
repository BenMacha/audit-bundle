JavaScript Code Example

// Audit data fetcher
class AuditDataFetcher {
    constructor(apiUrl) {
        this.apiUrl = apiUrl;
        this.cache = new Map();
    }
    
    async fetchAuditData(entityId) {
        if (this.cache.has(entityId)) {
            return this.cache.get(entityId);
        }
        
        try {
            const response = await fetch(`${this.apiUrl}/audit/${entityId}`);
            const data = await response.json();
            
            this.cache.set(entityId, data);
            return data;
        } catch (error) {
            console.error('Failed to fetch audit data:', error);
            throw error;
        }
    }
}
